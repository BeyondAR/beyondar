465  public int checkIntersectionPlane(Plane plane, Point3 position, Vector3 direction, double[] lamda,
466  Vector3 pNormal) {
467 
468  double DotProduct = direction.dotProduct(plane.getNormal());
469  double l2;
470 
471  // determine if ray parallel to plane
472  if ((DotProduct < ZERO) && (DotProduct > -ZERO))
473  return 0;
474 
475  Vector3 substract = new Vector3(plane.getPoint());
476  substract.subtract(position);
477  l2 = (plane.getNormal().dotProduct(substract)) / DotProduct;
478 
479  if (l2 < -ZERO)
480  return 0;
481 
482  pNormal.set(plane.getNormal());
483  lamda[0] = l2;
484  return 1;
485  }
