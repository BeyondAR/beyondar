157  private float fixedToFloat(int x) {
158  return x * (1.0f / 65536.0f);

161  private void preflight_adjust(int dir) {
162  int newTop = mTop + dir * MATRIX_SIZE;
163  if (newTop < 0) {
164  throw new IllegalArgumentException("stack underflow");
165  }
166  if (newTop + MATRIX_SIZE > mMatrix.length) {
167  throw new IllegalArgumentException("stack overflow");
168  }
169  }
170 
171  private void adjust(int dir) {
172  mTop += dir * MATRIX_SIZE;
173  }
