101  View view = getView(beyondarObject, toRecycle, mParentView);
102 
103  boolean added = false;
104  // Check if the recyclable view has been used, otherwise add
105  // it to the queue to recycle it
106  if ((toRecycle != view || view == null) && toRecycle != null) {
107  // Store it again to recycle it
108  mReusedViews.add(recycledParent);
109  added = true;
110  }
111 
112  // Check if the view has a parent, if not create it
113  if (view != null && (recycledParent == null || view.getParent() != recycledParent)) {
114  CustomLayout parentLayout = new CustomLayout(mContext);
115  parentLayout.addView(view, mLayoutParams);
116  if (!added) {
117  mReusedViews.add(recycledParent);
118  }
119  recycledParent = parentLayout;
120  }
121 
122  if (view != null) {
123  mNewViews.add(recycledParent);
124  if (recycledParent.getParent() == null) {
125  mParentView.addView(recycledParent, mLayoutParams);
