144  private synchronized int cleanRecycledBitmaps() {
145  int counter = 0;
146 
147  for (Map.Entry<String, Bitmap> entry : map.entrySet()) {
148  if (entry.getValue().isRecycled()) {
149  keysToClean.add(entry.getKey());
150  }
151  }
152 
153  // A second loop to avoid the concurrent modification exception
154  // while removing the elements
155  for (int i = 0; i < keysToClean.size(); i++) {
156  remove(keysToClean.get(i));
157  counter++;
158  }
159 
160  keysToClean.clear();
161 
162  if (DEBUG_CACHE) {
163  cacheLogD(mIdCache + " --- Recycled bitmaps cleaned: " + counter + "/"
164  + (map.size() + counter));
165  }
166  return counter;
167  }

275  public static int calculateInSampleSize(BitmapFactory.Options options, int reqWidth,
276  int reqHeight) {
277  // Raw height and width of image
278  final int height = options.outHeight;
279  final int width = options.outWidth;
280  int inSampleSize = 1;
281 
282  if (height > reqHeight || width > reqWidth) {
283  if (width > height) {
284  inSampleSize = Math.round((float) height / (float) reqHeight);
285  } else {
286  inSampleSize = Math.round((float) width / (float) reqWidth);
287  }
288  }
289  return inSampleSize;
290  }

374  options.inSampleSize = calculateInSampleSize(options, mMaxWidth, mMaxHeight);
375  options.inJustDecodeBounds = false;
376  }
377 
378  is.close();
379  is = assets.open(path);
380  btm = DebugBitmap.decodeStream(is, null, options, path);
381 
382  } catch (IOException e) {
383  e.printStackTrace();
384  }
385 
386  } else if (uri.startsWith(HEADER_HTTP) || uri.startsWith(HEADER_HTTPS)) {
387  btm = requestDownloadBitmap(uri);

559  @Override

562  Bitmap btm = null;

569  url = new URL(mUri);

583  options.inJustDecodeBounds = false;
584  }

586  is.close();
587  is = url.openStream();
588  btm = DebugBitmap.decodeStream(is, null, options, mUri);
589 
590  } catch (MalformedURLException e) {

592  cacheLogE(strData);

594  } catch (IOException e) {
595  String strData = mIdCache + " Error getting the image form internet: " + e;
596  cacheLogE(strData);

598  }
599  mBtm = btm;
600

602  storeBitmap(mUri, mBtm);
603  cacheLogD(mIdCache + " Image loaded form internet ("
604  + (System.currentTimeMillis() - time) + " ms): " + mUri);
605 
606  if (mOnLoadBitmapListener != null) {
607  for (int i = 0; i < mOnLoadBitmapListener.size(); i++) {
608  mOnLoadBitmapListener.get(i).onExternalBitmapLoaded(BitmapCache.this, mUri,
609  mBtm);
610  }
611  }
612 
613  } else {
