67  void registerSensor() {
68  if (isListenerRegistered || mSensorManager == null)
69  return;
70  List<Sensor> listSensors = mSensorManager.getSensorList(Sensor.TYPE_ACCELEROMETER);
71  if (listSensors.size() > 0) {
72  mSensorManager.registerListener(INSTANCE, listSensors.get(0), SENSOR_DELAY);
73  }
74 
75  listSensors = mSensorManager.getSensorList(Sensor.TYPE_MAGNETIC_FIELD);
76  if (listSensors.size() > 0) {
77  mSensorManager.registerListener(INSTANCE, listSensors.get(0), SENSOR_DELAY);
78  }
79  }
80 
81  void unregisterSensor() {
82  if (!isListenerRegistered)
83  return;
84  List<Sensor> listSensors = mSensorManager.getSensorList(Sensor.TYPE_ACCELEROMETER);
85  if (listSensors.size() > 0) {
86  mSensorManager.unregisterListener(INSTANCE, listSensors.get(0));
87  }
88 
89  listSensors = mSensorManager.getSensorList(Sensor.TYPE_MAGNETIC_FIELD);
90  if (listSensors.size() > 0) {
91  mSensorManager.unregisterListener(INSTANCE, listSensors.get(0));
92  }
93  }

117  @Override
118  public void onSensorChanged(SensorEvent event) {
119  int type = event.sensor.getType();
120  float[] values = null;
121 
122  switch (type) {
123  case Sensor.TYPE_ACCELEROMETER:
124  LowPassFilter.filter(event.values, mAccelerometerValues);
125  values = mAccelerometerValues;
126  break;
127  case Sensor.TYPE_MAGNETIC_FIELD:
128  LowPassFilter.filter(event.values, mMagneticValues);
129  values = mMagneticValues;
130  break;
131  default:
132  break;
133  }
134  if (values == null)
135  return;
136 
137  synchronized (mLock) {
138  for (int i = 0; i < mSensorsListener.size(); i++) {
139  mSensorsListener.get(i).onSensorChanged(values, event);
140  }
141  }
142  }
143  }
