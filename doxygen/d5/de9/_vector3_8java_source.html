178  /**

183  */
184  public int getGreatestComponent() {
185  int c = 0;
186  if (Math.abs(y) > Math.abs(x) && Math.abs(y) > Math.abs(z))
187  c = 1;
188  else if (Math.abs(z) > Math.abs(y) && Math.abs(z) > Math.abs(x))
189  c = 2;
190  return c;
191  }
