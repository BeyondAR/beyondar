96  private void calculateImageSizeRate() {
97  if (mWidth < mHeight) {
98  mWidthRate = ((float) mWidth / (float) mHeight);
99  mHeightRate = 1;
100  } else {
101  mHeightRate = ((float) mHeight / (float) mWidth);
102  mWidthRate = 1;
103  }
104 
105  for (int i = 0; i < mVertices.length; i++) {
106  if ((i + 1) % 3 == 0) {
107  mVertices[i] = mVertices[i] * mHeightRate;
108  } else {
109  mVertices[i] = mVertices[i] * mWidthRate;
110  }
111  }
112 
113  // a float has 4 bytes so we allocate for each coordinate 4 bytes
114  ByteBuffer byteBuffer = ByteBuffer.allocateDirect(mVertices.length * 4);
115  byteBuffer.order(ByteOrder.nativeOrder());
116  // allocates the memory from the byte buffer
117  mVertexBuffer = byteBuffer.asFloatBuffer();
118  // fill the vertexBuffer with the vertices
119  mVertexBuffer.put(mVertices);
120  // set the cursor position to the beginning of the buffer
121  mVertexBuffer.position(0);
122 
123  ByteBuffer byteBuffer2 = ByteBuffer.allocateDirect(mTextureMap.length * 4);
124  byteBuffer2.order(ByteOrder.nativeOrder());
125  mTextureBuffer = byteBuffer2.asFloatBuffer();
126  mTextureBuffer.put(mTextureMap);
127  mTextureBuffer.position(0);
128 
129  }
