77  private synchronized void checkResults() {
78  status++;
79 
80  if (status == 2 && callback != null) {
81 
82  if (btmCamera == null) {
83  callback.onScreenshot(btmGl);
84  return;
85  }
86  if (btmGl == null) {
87  callback.onScreenshot(btmCamera);
88  return;
89  }
90 
91  float factor = ((float) btmGl.getWidth() / (float) btmCamera.getWidth());
92 
93  float newWidth = factor * btmCamera.getWidth();
94  float newHeight = factor * btmCamera.getHeight();
95 
96  Bitmap newBtmCamera = ImageUtils.resizeImage(btmCamera, (int) newWidth, (int) newHeight);
97  if (newBtmCamera != btmCamera) {
98  btmCamera.recycle();
99  }
100 
101  Bitmap btm = ImageUtils.mergeBitmaps(newBtmCamera, btmGl);
102  newBtmCamera.recycle();
103 
104  Bitmap result = Bitmap.createBitmap(btm, 0, 0, btmGl.getWidth(), btmGl.getHeight());
105 
106  btmGl.recycle();
107  btm.recycle();
108 
109  System.gc();

111  }
