40  * Prepare the image to draw it using OpenGL. There are devices that doesn't

44  * @param bitmap

47  public static Bitmap resizeImageToPowerOfTwo(Bitmap bitmap) {
48  int width = bitmap.getWidth();
49  int height = bitmap.getHeight();
50 
51  int newWidth = (int) Math.pow(2, Math.ceil(Math.log10(width) / Math.log10(2)));
52  int newHeight = (int) Math.pow(2, Math.ceil(Math.log10(height) / Math.log10(2)));
53 
54  if (newWidth > MAX_SIZE) {
55  newWidth = MAX_SIZE;
56  }
57 
58  if (newHeight > MAX_SIZE) {
59  newHeight = MAX_SIZE;
60  }
61 
62  // calculate the scale - in this case = 0.4f
63  float scaleWidth = ((float) newWidth) / width;
64  float scaleHeight = ((float) newHeight) / height;
65 
66  // create a matrix for the manipulation
67  Matrix matrix = new Matrix();
68  // resize the bit map
69  matrix.postScale(scaleWidth, scaleHeight);
70  // rotate the Bitmap
71  // matrix.postRotate(45);
72 
73  // recreate the new Bitmap
74  return Bitmap.createBitmap(bitmap, 0, 0, width, height, matrix, true);
