189  /**
190  * This method try to find out what is the best size for previewing the
191  * camera.
192  *
193  * @param sizes
194  * @param width
195  * @param height
196  * @return
197  */
198  private Size getOptimalPreviewSize(List<Size> sizes, int width, int height) {
199  double targetRatio = (double) width / height;
200 
201  if (sizes == null)
202  return null;
203 
204  Camera.Size optimalSize = null;
205  double minDiff = Double.MAX_VALUE;
206 
207  int targetHeight = height;
208 
209  for (Camera.Size size : sizes) {
210  double ratio = (double) size.width / size.height;
211  if (Math.abs(ratio - targetRatio) > ASPECT_TOLERANCE) {
212  continue;
213  }
214  if (Math.abs(size.height - targetHeight) < minDiff) {
215  optimalSize = size;
216  minDiff = Math.abs(size.height - targetHeight);
217  }
218  }
219 
220  if (optimalSize == null) {
221  minDiff = Double.MAX_VALUE;
222  for (Camera.Size size : sizes) {
223  if (Math.abs(size.height - targetHeight) < minDiff) {
224  optimalSize = size;
225  minDiff = Math.abs(size.height - targetHeight);
226  }
227  }
228  }
229  return optimalSize;
230  }

270  @TargetApi(Build.VERSION_CODES.GINGERBREAD)
271  private int getCameraDisplayOrientation() {
272  int rotation = ((WindowManager) getContext().getSystemService(Context.WINDOW_SERVICE))
273  .getDefaultDisplay().getRotation();
274  int degrees = 0;
275  switch (rotation) {
276  case Surface.ROTATION_0:
277  degrees = 0;
278  break;
279  case Surface.ROTATION_90:
280  degrees = 90;
281  break;
282  case Surface.ROTATION_180:
283  degrees = 180;
284  break;
285  case Surface.ROTATION_270:
286  degrees = 270;
287  break;
288  }
289 
290  int result = 0;
291  android.hardware.Camera.CameraInfo info = new android.hardware.Camera.CameraInfo();
292  android.hardware.Camera.getCameraInfo(Camera.CameraInfo.CAMERA_FACING_BACK, info);
293  if (info.facing == Camera.CameraInfo.CAMERA_FACING_FRONT) {
294  result = (info.orientation + degrees) % 360;
295  result = (360 - result) % 360; // compensate the mirror
296  } else { // back-facing
297  result = (info.orientation - degrees + 360) % 360;
298  }
299  return result;
300  }

388  private boolean openCamera() {
389  Logger.v(TAG, "getTheCamera");
390  // keep trying to acquire the camera until "maximumWaitTimeForCamera"
391  // seconds have passed
392  boolean acquiredCam = false;
393  int timePassed = 0;
394  while (!acquiredCam && timePassed < MAX_TIME_WAIT_FOR_CAMERA) {
395  try {
396  mCamera = Camera.open();
397  Logger.v(TAG, "acquired the camera");
398  acquiredCam = true;
399  return true;
400  } catch (Exception e) {
401  Logger.e(TAG, "Exception encountered opening camera:" + e.getLocalizedMessage());
402  }
403  try {
404  Thread.sleep(200);
405  } catch (InterruptedException ee) {
406  Logger.e(TAG, "Exception encountered sleeping:" + ee.getLocalizedMessage());
407  }
408  timePassed += 200;
409  }
410  return false;
411  }
