75  /** The draw method for the square with the GL context */
76  @Override
77  public void draw(GL10 gl, Texture defaultTexture) {
78 
79  mTexture = mBeyondarObject.getTexture();
80  Texture texture = mTexture;
81 
82  gl.glTranslatef(mPosition.x, mPosition.y, mPosition.z);
83 
84  // ROTATE According to the angles
85 
86  gl.glRotatef((float) mAngle.x, 1, 0, 0);
87  gl.glRotatef((float) mAngle.y, 0, 1, 0);
88  gl.glRotatef((float) mAngle.z, 0, 0, 1);
89 
90  // bind the previously generated texture
91  if (!mTexture.isLoaded()) {
92  gl.glBindTexture(GL10.GL_TEXTURE_2D, defaultTexture.getTexturePointer());
93  texture = defaultTexture;
94  } else {
95  gl.glBindTexture(GL10.GL_TEXTURE_2D, mTexture.getTexturePointer());
96  }
97 
98  // Point to our buffers
99  gl.glEnableClientState(GL10.GL_VERTEX_ARRAY);
100  gl.glEnableClientState(GL10.GL_TEXTURE_COORD_ARRAY);
101 
102  // Set the face rotation
103  gl.glFrontFace(GL10.GL_CW);
104 
105  // Point to our vertex buffer
106  gl.glVertexPointer(3, GL10.GL_FLOAT, 0, texture.getVerticesBuffer());
107  gl.glTexCoordPointer(2, GL10.GL_FLOAT, 0, texture.getTextureBuffer());
108 
109  // Draw the vertices as triangle strip
110  gl.glDrawArrays(GL10.GL_TRIANGLE_STRIP, 0, texture.getVertices().length / 3);
111 
112  // Disable the client state before leaving
113  gl.glDisableClientState(GL10.GL_VERTEX_ARRAY);
114  gl.glDisableClientState(GL10.GL_TEXTURE_COORD_ARRAY);
115 
116  // gl.glPopMatrix();
117 
118  // rotate to the previous state
119  gl.glRotatef((float) mAngle.x, -1, 0, 0);
120  gl.glRotatef((float) mAngle.y, 0, -1, 0);
121  gl.glRotatef((float) mAngle.z, 0, 0, -1);
122 
123  gl.glTranslatef(-mPosition.x, -mPosition.y, -mPosition.z);
124  }
