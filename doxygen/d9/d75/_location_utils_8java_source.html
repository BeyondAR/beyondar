20 public class LocationUtils {
21  private static final int TWO_MINUTES = 1000 * 60 * 2;
22 
23  /**
24  * Determines whether one Location reading is better than the current
25  * Location fix
26  *
27  * @param location
28  * The new Location that you want to evaluate
29  * @param currentBestLocation
30  * The current Location fix, to which you want to compare the new
31  * one
32  */
33  public static boolean isBetterLocation(Location location, Location currentBestLocation) {
34  if (location == null) {
35  return false;
36  }
37  if (currentBestLocation == null) {
38  // A new location is always better than no location
39  return true;
40  }
41 
42  // Check whether the new location fix is newer or older
43  long timeDelta = location.getTime() - currentBestLocation.getTime();
44  boolean isSignificantlyNewer = timeDelta > TWO_MINUTES;
45  boolean isSignificantlyOlder = timeDelta < -TWO_MINUTES;
46  boolean isNewer = timeDelta > 0;
47 
48  // If it's been more than two minutes since the current location, use
49  // the new location
50  // because the user has likely moved
51  if (isSignificantlyNewer) {
52  return true;
53  // If the new location is more than two minutes older, it must be
54  // worse
55  } else if (isSignificantlyOlder) {
56  return false;
57  }
58 
59  // Check whether the new location fix is more or less accurate
60  int accuracyDelta = (int) (location.getAccuracy() - currentBestLocation.getAccuracy());
61  boolean isLessAccurate = accuracyDelta > 0;
62  boolean isMoreAccurate = accuracyDelta < 0;
63  boolean isSignificantlyLessAccurate = accuracyDelta > 200;
64 
65  // Check if the old and new location are from the same provider
66  boolean isFromSameProvider = isSameProvider(location.getProvider(), currentBestLocation.getProvider());
67 
68  // Determine location quality using a combination of timeliness and
69  // accuracy
70  if (isMoreAccurate) {
71  return true;
72  } else if (isNewer && !isLessAccurate) {
73  return true;
74  } else if (isNewer && !isSignificantlyLessAccurate && isFromSameProvider) {
75  return true;
76  }
77  return false;
78  }
79 
80  /** Checks whether two providers are the same */
81  public static boolean isSameProvider(String provider1, String provider2) {
82  if (provider1 == null) {
83  return provider2 == null;
84  }
85  return provider1.equals(provider2);
86  }
