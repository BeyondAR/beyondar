24 /**
25  * A cache that holds strong references to a limited number of values. Each time
26  * a value is accessed, it is moved to the head of a queue. When a value is
27  * added to a full cache, the value at the end of that queue is evicted and may
28  * become eligible for garbage collection.
29  *
30  * <p>
31  * If your cached values hold resources that need to be explicitly released,
32  * override {@link #entryEvicted}. This method is only invoked when values are
33  * evicted. Values replaced by calls to {@link #put} must be released manually.
34  *
35  * <p>
36  * If a cache miss should be computed on demand for the corresponding keys,
37  * override {@link #create}. This simplifies the calling code, allowing it to
38  * assume a value will always be returned, even when there's a cache miss.
39  *
40  * <p>
41  * By default, the cache size is measured in the number of entries. Override
42  * {@link #sizeOf} to size the cache in different units. For example, this cache
43  * is limited to 4MiB of bitmaps:
44  *
45  * <pre>
46  * {@code
47  * int cacheSize = 4 * 1024 * 1024; // 4MiB
48  * LruCache<String, Bitmap> bitmapCache = new LruCache<String, Bitmap>(cacheSize) {
49  * protected int sizeOf(String key, Bitmap value) {
50  * return value.getByteCount();
51  * }
52  * }}
53  * </pre>
54  *
55  * <p>
56  * This class is thread-safe. Perform multiple cache operations atomically by
57  * synchronizing on the cache:
58  *
59  * <pre>
60  * {@code
61  * synchronized (cache) {
62  * if (cache.get(key) == null) {
63  * cache.put(key, value);
64  * }
65  * }}
66  * </pre>
67  *
68  * <p>
69  * This class does not allow null to be used as a key or value. A return value
70  * of null from {@link #get}, {@link #put} or {@link #remove} is unambiguous:
71  * the key was not in the cache.
72  */

88  * for caches that do not override {@link #sizeOf}, this is the

90  * this is the maximum sum of the sizes of the entries in this
91  * cache.

93  public LruCache(int maxSize) {
94  if (maxSize <= 0) {
95  throw new IllegalArgumentException("maxSize <= 0");
96  }
97  this.maxSize = maxSize;
98  this.map = new LinkedHashMap<K, V>(0, 0.75f, true);

101  /**
102  * Returns the value for {@code key} if it exists in the cache or can be
103  * created by {@code #create}. If a value was returned, it is moved to the
104  * head of the queue. This returns null if a value is not cached and cannot
105  * be created.
106  */
107  public synchronized final V get(K key) {
108  if (key == null) {
109  throw new NullPointerException("key == null");
110  }
111 
112  V result = map.get(key);
113  if (result != null) {
114  hitCount++;
115  return result;
116  }
117 
118  missCount++;
119 
120  result = create(key);
121 
122  if (result != null) {
123  createCount++;
124  size += safeSizeOf(key, result);
125  map.put(key, result);
126  trimToSize(maxSize);
127  }
128  return result;
129  }
130 
131  /**
132  * Caches {@code value} for {@code key}. The value is moved to the head of
133  * the queue.
134  *
135  * @return the previous value mapped by {@code key}. Although that entry is
136  * no longer cached, it has not been passed to {@link #entryEvicted}
137  * .
138  */
139  public synchronized final V put(K key, V value) {
140  if (key == null || value == null) {
141  throw new NullPointerException("key == null || value == null");
142  }
143 
144  putCount++;
145  size += safeSizeOf(key, value);
146  V previous = map.put(key, value);
147  if (previous != null) {
148  size -= safeSizeOf(key, previous);
149  }
150  trimToSize(maxSize);
151  return previous;
152  }
153 
154  protected void trimToSize(int maxSize) {
155  while (size > maxSize) {
156  Map.Entry<K, V> toEvict = map.entrySet().iterator().next();
157  if (toEvict == null) {
158  break; // map is empty; if size is not 0 then throw an error
159  // below
160  }
161 
162  K key = toEvict.getKey();
163  V value = toEvict.getValue();
164  map.remove(key);
165  size -= safeSizeOf(key, value);
166  evictionCount++;
167 
168  entryEvicted(key, value);
169  }
170 
171  if (size < 0 || (map.isEmpty() && size != 0)) {
172  throw new IllegalStateException(getClass().getName()
173  + ".sizeOf() is reporting inconsistent results!");
174  }
175  }
176 
177  /**
178  * Removes the entry for {@code key} if it exists.
179  *
180  * @return the previous value mapped by {@code key}. Although that entry is
181  * no longer cached, it has not been passed to {@link #entryEvicted}
182  * .
183  */
184  public synchronized final V remove(K key) {
185  if (key == null) {
186  throw new NullPointerException("key == null");
187  }
188 
189  V previous = map.remove(key);
190  if (previous != null) {
191  size -= safeSizeOf(key, previous);
192  }
193 
194  entryRemoved(key, previous);
195  return previous;
196  }
