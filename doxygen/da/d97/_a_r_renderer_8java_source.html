336  /**
337  * Generate a new bitmap from the gl.
338  *
339  * @param gl
340  * @return A new Bitmap of the last frame.
341  */
342  protected Bitmap savePixels(GL10 gl) {
343  // http://stackoverflow.com/questions/3310990/taking-screenshot-of-android-opengl
344  // http://www.anddev.org/how_to_get_opengl_screenshot__useful_programing_hint-t829.html
345  int b[] = new int[mWidth * mHeight];
346  IntBuffer ib = IntBuffer.wrap(b);
347  ib.position(0);
348  gl.glReadPixels(0, 0, mWidth, mHeight, GL10.GL_RGBA, GL10.GL_UNSIGNED_BYTE, ib);
349 
350  // The bytes within the ints are in the wrong order for android, but
351  // convert into a
352  // bitmap anyway. They're also bottom-to-top rather than top-to-bottom.
353  // We'll fix
354  // this up soon using some fast API calls.
355  Bitmap glbitmap = Bitmap.createBitmap(b, mWidth, mHeight, Bitmap.Config.ARGB_4444);
356  ib = null; // we're done with ib
357  b = null; // we're done with b, so allow the memory to be freed
358 
359  // To swap the color channels, we'll use a
360  // ColorMatrix/ColorMatrixFilter. From the Android docs:
361  //
362  // This is a 5x4 matrix: [ a, b, c, d, e, f, g, h, i, j, k, l, m, n, o,
363  // p, q, r, s, t ]
364  // When applied to a color [r, g, b, a] the resulting color is computed
365  // as (after clamping):
366  //
367  // R' = a*R + b*G + c*B + d*A + e;
368  // G' = f*R + g*G + h*B + i*A + j;
369  // B' = k*R + l*G + m*B + n*A + o;
370  // A' = p*R + q*G + r*B + s*A + t;
371  //
372  // We want to swap R and B, so the coefficients will be:
373  // R' = B => 0,0,1,0,0
374  // G' = G => 0,1,0,0,0
375  // B' = R => 1,0,0,0,0
376  // A' = A => 0,0,0,1,0
377 
378  final float[] cmVals = { 0, 0, 1, 0, 0, //
379  0, 1, 0, 0, 0, //
380  1, 0, 0, 0, 0, //
381  0, 0, 0, 1, 0 };
382 
383  Paint paint = new Paint();
384 
385  // our R<->B swapping paint
386  paint.setColorFilter(new ColorMatrixColorFilter(new ColorMatrix(cmVals)));
387 
388  // the bitmap we're going to draw onto
389  Bitmap bitmap = Bitmap.createBitmap(mWidth, mHeight, Config.ARGB_4444);
390 
391  // we draw to the bitmap through a canvas
392  Canvas canvas = new Canvas(bitmap);
393  // draw the opengl bitmap onto the canvas, using the color swapping
394  // paint
395  canvas.drawBitmap(glbitmap, 0, 0, paint);
396 
397  glbitmap.recycle();
398  glbitmap = null; // we're done with glbitmap, let go of its memory
399 
400  // the image is still upside-down, so vertically flip it
401  Matrix matrix = new Matrix();
402  matrix.preScale(1.0f, -1.0f); // scaling: x = x, y = -y, i.e. vertically
403  // flip
404 
405  // new bitmap, using the flipping matrix
406  Bitmap result = Bitmap
407  .createBitmap(bitmap, 0, 0, bitmap.getWidth(), bitmap.getHeight(), matrix, true);
408 
409  bitmap.recycle();
410  System.gc();
411  return result;
412  }

416  /**
417  * Convert the geolocation to gl points representation.
418  *
419  * @param geoObject
420  * The {@link com.beyondar.android.world.GeoObject GeoObject} to
421  * be processed.
422  * @param out
423  * Where the results be stored.
424  */
425  protected void convertGPStoPoint3(GeoObject geoObject, Point3 out) {
426  float x, z, y;
427  x = (float) (Distance.fastConversionGeopointsToMeters(geoObject.getLongitude()
428  - mWorld.getLongitude()) / 2);
429  z = (float) (Distance.fastConversionGeopointsToMeters(geoObject.getAltitude() - mWorld.getAltitude()) / 2);
430  y = (float) (Distance.fastConversionGeopointsToMeters(geoObject.getLatitude() - mWorld.getLatitude()) / 2);
431 
432  if (mMaxDistanceSizePoints > 0 || mMinDistanceSizePoints > 0) {
433  double totalDst = Distance.calculateDistance(x, y, 0, 0);
434 
435  if (mMaxDistanceSizePoints > 0 && totalDst > mMaxDistanceSizePoints) {
436  MathUtils.linearInterpolate(0, 0, 0, x, y, 0, mMaxDistanceSizePoints, totalDst, sOut);
437  x = (float) sOut[0];
438  y = (float) sOut[1];
439  if (mMinDistanceSizePoints > 0) {
440  totalDst = Distance.calculateDistance(x, y, 0, 0);
441  }
442  }
443  if (mMinDistanceSizePoints > 0 && totalDst < mMinDistanceSizePoints) {
444  MathUtils.linearInterpolate(0, 0, 0, x, y, 0, mMinDistanceSizePoints, totalDst, sOut);
445  x = (float) sOut[0];
446  y = (float) sOut[1];
447  }
448  }
449 
450  out.x = x;
451  out.y = y;
452  out.z = z;
453  }

612  /**
613  * Render a specific {@link com.beyondar.android.world.BeyondarObject
614  * BeyondarObject}.
615  *
616  * @param gl
617  * @param beyondarObject
618  * {@link com.beyondar.android.world.BeyondarObject
619  * BeyondarObject} to render.
620  * @param defaultTexture
621  * If the texture from the
622  * {@link com.beyondar.android.world.BeyondarObject
623  * BeyondarObject} is not available the defaultTexure will be
624  * used.
625  */
626  protected void renderBeyondarObject(GL10 gl, BeyondarObject beyondarObject, Texture defaultTexture,
627  long time) {
628  boolean renderObject = false;
629  Renderable renderable = beyondarObject.getOpenGLObject();
630 
631  if (renderable == null || !beyondarObject.isVisible()) {
632  return;
633  }
634  double dst = 0;
635  if (beyondarObject instanceof GeoObject) {
636  dst = ((GeoObject) beyondarObject).calculateDistanceMeters(mWorld.getLongitude(),
637  mWorld.getLatitude());
638  convertGPStoPoint3((GeoObject) beyondarObject, beyondarObject.getPosition());
639  } else {
640  Point3 position = beyondarObject.getPosition();
641  dst = MathUtils.GLUnitsToMeters((float) Distance.calculateDistanceCoordinates(0, 0, 0,
642  position.x, position.y, position.z));
643  }
644 
645  beyondarObject.setDistanceFromUser(dst);
646 
647  if (dst < mWorld.getArViewDistance()) {
648  renderObject = true;
649  }
650 
651  boolean forceDraw = renderable.update(time, (float) dst, beyondarObject);
652 
653  if (forceDraw || renderObject) {
654  if (beyondarObject.isFacingToCamera()) {
655  MathUtils.calcAngleFaceToCamera(beyondarObject.getPosition(), mCameraPosition,
656  beyondarObject.getAngle());
657  }
658 
659  if (!beyondarObject.getTexture().isLoaded() && beyondarObject.getImageUri() != null) {
660  int counter = beyondarObject.getTexture().getLoadTryCounter();
661  double timeOut = TIMEOUT_LOAD_TEXTURE * (counter + 1);
662  if (beyondarObject.getTexture().getTimeStamp() == 0

958  /**
959  * Create the texture for the specified Bitmap.<br>
960  * __Important__ The method will recycle the bitmap after being used.
961  *
962  * @param gl
963  * @param bitmap
964  * @return A new texture for the bitmap.
965  */
966  public Texture load2DTexture(GL10 gl, Bitmap bitmap) {
967  // see
968  // http://stackoverflow.com/questions/3921685/issues-with-glutils-teximage2d-and-alpha-in-textures
969  int[] tmpTexture = new int[1];
970  if (null == bitmap) {
971  return null;
972  }
973  int imageWidth = bitmap.getWidth();
974  int imageHeight = bitmap.getHeight();
975 
976  if (!isGL_OES_texture_npot && !Utils.isCompatibleWithOpenGL(bitmap)) {
977  Bitmap tmp = Utils.resizeImageToPowerOfTwo(bitmap);
978  bitmap.recycle();
979  bitmap = tmp;
980  }
981 
982  // generate one texture pointer
983  gl.glGenTextures(1, tmpTexture, 0);
984  // ...and bind it to our array
985  gl.glBindTexture(GL10.GL_TEXTURE_2D, tmpTexture[0]);
986 
987  // create nearest filtered texture
988  gl.glTexParameterf(GL10.GL_TEXTURE_2D, GL10.GL_TEXTURE_MIN_FILTER, GL10.GL_NEAREST);
989  gl.glTexParameterf(GL10.GL_TEXTURE_2D, GL10.GL_TEXTURE_MAG_FILTER, GL10.GL_LINEAR);
990 
991  // Different possible texture parameters, e.g.
992  // GL10.GL_CLAMP_TO_EDGE
993  gl.glTexParameterf(GL10.GL_TEXTURE_2D, GL10.GL_TEXTURE_WRAP_S, GL10.GL_REPEAT);
994  gl.glTexParameterf(GL10.GL_TEXTURE_2D, GL10.GL_TEXTURE_WRAP_T, GL10.GL_REPEAT);
995  //
996  // Use Android GLUtils to specify a two-dimensional
997  // texture image from our bitmap
998  GLUtils.texImage2D(GL10.GL_TEXTURE_2D, 0, bitmap, 0);
999  //
1000  // Clean up
1001  bitmap.recycle();
1002  return new Texture(tmpTexture[0]).setImageSize(imageWidth, imageHeight);
1003 
1004  }
