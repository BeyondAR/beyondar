57  * removed from the 3D point
58  *
59  * @param p
60  * 3D point
61  * @param remove
62  * coordinate index to be removed. If 0, x will be removed,
63  * assigning 2D x to 3D y and 2D y to 3D z. If 1, y will be
64  * removed, and so on... If <em>remove</em> if greater than 2 or
65  * less than 0, the z coordinate will be removed
66  */
67  public Point2(Point3 p, int remove) {
68  switch (remove) {
69  case 0:
70  x = p.y;
71  y = p.z;
72  break;
73  case 1:
74  x = p.x;
75  y = p.z;
76  break;
77  default:
78  x = p.x;
79  y = p.y;
80  }
81  }
