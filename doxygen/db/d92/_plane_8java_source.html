73  /**
74  * Returns the t parameter for the intersection with the given ray. If plane
75  * and ray has no intersection returns <b>-1</b> representing infinite.
76  * Value returned must be used with {@link Ray#getPoint(float)}
77  *
78  * @param ray
79  * Ray to be checked
80  * @return the t parameter for the ray in the intersection point. If plane
81  * and ray has no intersection returns <b>-1</b>
82  */
83  public float intersects(Ray ray) {
84  float dotProduct = ray.getVector().dotProduct(normal);
85  if (Math.abs(dotProduct) > 0.01f) {
86  Point3 p = ray.getPoint();
87  float t = -(p.x * normal.x + p.y * normal.y + p.z * normal.z + d)
88  / dotProduct;
89  return t;
90  } else
91  return -1;
92  }
