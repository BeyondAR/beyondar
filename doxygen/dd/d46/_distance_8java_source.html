82  public static double calculateDistanceMeters(double aLong, double aLat, double bLong, double bLat) {
83 
84  double d2r = (Math.PI / 180);
85 
86  double dLat = (bLat - aLat) * d2r;
87  double dLon = (bLong - aLong) * d2r;
88  double a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos(aLat * d2r) * Math.cos(bLat * d2r)
89  * Math.sin(dLon / 2) * Math.sin(dLon / 2);
90  double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
91 
92  return EARTH_RADIUS_KM * c * 1000;
93 
94  }
