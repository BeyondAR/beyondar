69  /**

71  *
72  * @param context
73  * @return true if it is >= 6.5 and < 7.6
74  */
75  public static boolean is7InchTablet(Context context) {
76  DisplayMetrics metrics = context.getResources().getDisplayMetrics();
77 
78  float widthInInches = metrics.widthPixels / metrics.xdpi;
79  float heightInInches = metrics.heightPixels / metrics.ydpi;
80 
81  double sizeInInches = Math.sqrt(Math.pow(widthInInches, 2) + Math.pow(heightInInches, 2));
82  return sizeInInches >= 6.5 && sizeInInches < 7.6;

84  }
