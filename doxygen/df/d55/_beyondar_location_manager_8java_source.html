28  * application needs to add the following permissions in the manifest: <br>

35  * Remember that you also can use the Location utility in the Google Services. <br>
36  * <br>
37  * Here is a small example how to use {@link BeyondarLocationManager}: <br>
38  *
39  * <pre>
40  * <code>
41  * void onCreate(Bundle savedInstanceState){
42  * BeyondarLocationManager.setLocationManager((LocationManager) this.getSystemService(Context.LOCATION_SERVICE));
43  * BeyondarLocationManager.addGeoObjectLocationUpdate(beyondarObject);
44  * // You also can register a World or a LocationListener

84  public void enable() {
85  if (mLocationManager == null) {
86  mIsEnabled = false;
87  return;
88  }
89  if (!mIsEnabled) {
90  Location lastGpsLocation = mLocationManager
91  .getLastKnownLocation(LocationManager.GPS_PROVIDER);
92  Location lastNetworkLocation = mLocationManager
93  .getLastKnownLocation(LocationManager.NETWORK_PROVIDER);
94 
95  if (LocationUtils.isBetterLocation(lastNetworkLocation, lastGpsLocation)) {
96  mLocationListener.setLastKnowLocation(lastNetworkLocation);
97  } else {
98  mLocationListener.setLastKnowLocation(lastGpsLocation);
99  }
100  }
101  registerLocationListener(mLocationListener, mGpsListener);
102 
103  mIsEnabled = true;
104 
105  }
106 
107  public void registerLocationListener(LocationListener locationListener, GpsStatus.Listener gpsListener) {
108  if (mLocationManager.getAllProviders().contains(LocationManager.GPS_PROVIDER)) {
109  mLocationManager.requestLocationUpdates(LocationManager.GPS_PROVIDER, 0, 0, locationListener);
110  mLocationManager.addGpsStatusListener(gpsListener);
111  }
112  if (mLocationManager.getAllProviders().contains(LocationManager.NETWORK_PROVIDER)) {
113  mLocationManager.requestLocationUpdates(LocationManager.NETWORK_PROVIDER, 0, 0,
114  locationListener);
115  }
116 
117  }
118

120  mIsEnabled = false;

122  if (mLocationManager != null) {
123  mLocationManager.removeUpdates(mLocationListener);
124  }

135  private class BeyondarGpsListener implements GpsStatus.Listener {
136  public void onGpsStatusChanged(int event) {
137  switch (event) {
138  case GpsStatus.GPS_EVENT_SATELLITE_STATUS:
139  if (mLocationListener.getLastGpsLocation() != null) {
140  if ((System.currentTimeMillis() - mLocationListener.getLastGpsLocation().getTime()) < MAX_TIME_GPS_FIX) {
141  if (!mGpsFix)
142  mGpsFix = true;
143  } else {
144  if (mGpsFix) {
145  enable();
146  }
147  mGpsFix = false;
148  }
149  }
150  break;
151  case GpsStatus.GPS_EVENT_FIRST_FIX:
152  mGpsFix = true;
153  break;
154  case GpsStatus.GPS_EVENT_STARTED:
155  break;
156  case GpsStatus.GPS_EVENT_STOPPED:
157  mGpsFix = false;
158  break;
159  }
160  }
161  }
