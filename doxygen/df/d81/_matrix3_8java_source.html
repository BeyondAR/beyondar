100  /**
101  * Multiplies m1*m2, and deposit the result at mresult
102  *
103  * @param m1
104  * first matrix
105  * @param m2
106  * second matrix
107  * @param mresult
108  * result matrix
109  */
110  private void multiplyMatrix(float m1[], float m2[], float mresult[]) {
111  float matrix[] = new float[DIMENSION];
112  int row = 0, column = 0;
113 
114  for (int i = 0; i < 16; i++) {
115  row = i % 4;
116  column = i / 4;
117  matrix[i] = 0;
118  for (int j = 0; j < 4; j++) {
119  matrix[i] += m1[row + j * 4] * m2[column * 4 + j];
120  }
121  }
122 
123  for (int i = 0; i < 16; i++)
124  mresult[i] = matrix[i];
125  }

178  float my[] = new float[DIMENSION];
179  float mz[] = new float[DIMENSION];
180  float mi[] = new float[DIMENSION];

183  initRotationMatrixY(my, y);
184  initMatrixRotationZ(mz, z);

186  multiplyMatrix(mx, my, mi);
187  multiplyMatrix(mi, mz, mr);
188  postMultiply(mr);
